# Covariance
def cov(x, y):
  
    xbar, ybar = x.mean(), y.mean()
    # formula 1 
    F1 = np.sum((x - xbar)*(y - ybar))/(len(x) - 1)
    # formula 2
    F2 = np.dot((x - xbar), (y - ybar).T) / (len(x) - 1)

    return F2

# Covariance matrix
def cov_mat(X):
    return np.array([[cov(X[0], X[0]), cov(X[0], X[1])], \
                     [cov(X[1], X[0]), cov(X[1], X[1])]])
